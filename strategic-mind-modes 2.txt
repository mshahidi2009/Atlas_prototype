const recommendPrompt = `You are a strategic thinking coach. Given a scenario, recommend the 5 most relevant thinking modes from this list and explain in one sentence why each fits.

Modes: ${MODES.map(m => m.name).join(", ")}

Respond ONLY with valid JSON, no markdown, no backticks, no explanation outside JSON:
{"recommendations":[{"name":"Mode Name","reason":"One sentence why"}]}`;

function parseMarkdown(text) {
  return text
    .replace(/\*\*(.*?)\*\*/g, '<strong>$1</strong>')
    .replace(/^• (.+)$/gm, '<li>$1</li>')
    .replace(/(<li>.*<\/li>)/s, '<ul>$1</ul>')
    .replace(/\n\n/g, '<br/><br/>')
    .replace(/\n/g, '<br/>');
}
